import networkx as nx from graphviz import Graph #from IPython.display import display_html def diagram(g): P = Graph(engine='neato') P.attr('node', shape='circle', width='.15', height='.15', fixedsize='true', fontsize='10') P.edges([(str(i),str(j)) for i in range(len(g)) for j in g[i] if i>j]) return P def showg(li): #display a list of undirected graphs i = 0 st = "" for x in li: i+=1 st+=str(i) st+=diagram(x)._repr_svg_()+"        " display_html(st,raw=True) G = nx.complete_graph(5) nx.draw(G) #showg([ # [[1,2],[0,2],[0,1]], # [[1,2],[0],[0]] # ])
